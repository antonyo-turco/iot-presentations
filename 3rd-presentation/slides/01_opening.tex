\section{Problem}

\begin{frame}{The Problem in 10 Seconds}
\begin{center}
  \large\textbf{Detect intrusions through windows, without wiring, for months on a battery.}
\end{center}

\vspace{0.2cm}
\begin{columns}[T, onlytextwidth]
\begin{column}{0.48\textwidth}
  \textbf{Constraints}
  \begin{itemize}\setlength\itemsep{3pt}
    \item No wiring (renters, retrofit installs)
    \item Battery must last months, not days
    \item Sub-\texteuro10 per sensor node
    \item Must detect vibrations accurately from inside the window frame
  \end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
  \textbf{Requirements}
  \begin{itemize}\setlength\itemsep{3pt}
    \item Quiescent current $<100\,\mu$A
    \item Detection latency $<5\,\text{s}$
    \item Battery life $>6$ months
    \item False positive rate near zero (traffic, wind, rain unacceptable)
  \end{itemize}
\end{column}
\end{columns}
\end{frame}
