\section{Problem}

\begin{frame}{The Problem in 10 Seconds}
\begin{center}
  \large\textbf{Monitor a door 24/7, without wiring, for months on a battery.}
\end{center}

\vspace{0.15cm}
\begin{columns}[T, onlytextwidth]
\begin{column}{0.48\textwidth}
  \textbf{The constraints}
  \begin{itemize}\setlength\itemsep{2pt}
    \item No wiring --- renters, retrofit installs
    \item Battery must last months, not days
    \item False positives (traffic, wind) unacceptable
  \end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
  \textbf{Our approach}
  \begin{itemize}\setlength\itemsep{2pt}
    \item Sub-\texteuro10 sensor, $<$63\,$\mu$A idle
    \item On-device K-means learns the door signature
    \item Alarms only on genuine knock events
  \end{itemize}
\end{column}
\end{columns}

\vspace{0.15cm}
\begin{block}{Result}
  Detection in $<$3 s, $>$50 days on 2600 mAh, three independently encrypted links.
\end{block}
\end{frame}
