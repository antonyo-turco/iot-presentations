  \large
  \begin{itemize}

  \end{colorblock}
\end{column}
\end{columns}

% --- K-Means++ pipeline deep-dive ---
\begin{frame}[shrink=5]{On-Device K-Means++ Pipeline}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.31\textwidth}
  \begin{colorblock}[white]{sinteflilla}{\small\textbf{1. Training}}
  \scriptsize
  \begin{itemize}\setlength\itemsep{2pt}
    \item 512-pt FFT $\rightarrow$ 51-D feature vector
    \item \textbf{Online K-Means++}: K-Means++ init + incremental online updates
    \item 24h exploration phase on actual window
    \item No full dataset retained on device
  \end{itemize}
  \end{colorblock}
\end{column}
\begin{column}{0.31\textwidth}
  \begin{colorblock}[white]{maincolor}{\small\textbf{2. Threshold}}
  \scriptsize
  \begin{itemize}\setlength\itemsep{2pt}
    \item Per-cluster: $\theta_k = \mu_k + 3\sigma_k$
    \item Computed from training distribution
    \item $K = 10$ clusters
    \item Self-calibrates without retraining from scratch
  \end{itemize}
  \end{colorblock}
\end{column}
\begin{column}{0.31\textwidth}
  \begin{colorblock}[white]{sintefgreen}{\small\textbf{3. Inference}}
  \scriptsize
  \begin{itemize}\setlength\itemsep{2pt}
    \item Compute distance $d^*$ to nearest centroid
    \item $d^* \leq \theta_k$: \textbf{baseline} (ignore)
    \item $d^* > \theta_k$: \textbf{alarm}
    \item $O(K{\cdot}d)$: 510 MACs at $K{=}10$, $d{=}51$; $<10$\,ms on ESP32-C3
  \end{itemize}
  \end{colorblock}
\end{column}
\end{columns}
\vspace{0.3cm}
\begin{block}{Evaluation}
\scriptsize Tested across multiple installation positions on the same window frame; $\theta_k$ updated incrementally without full retraining.
\end{block}
\note{
  Isolation Forest was the first algorithm evaluated. Rejected for three reasons:
  (1) Isolation Forest requires batch training; it cannot update a forest
      incrementally without rebuilding from the full dataset, which cannot be
      retained on-device (insufficient RAM on ESP32-C3).
  (2) Cloud-based training was explored as a workaround, but each training update
      would require a Wi-Fi TX round-trip, which violates the energy budget and
      introduces dependency on internet availability.
  (3) K-Means++ inference is O(K x d): one Euclidean distance per centroid.
      With K=10 and d=51, this is 510 multiply-accumulates, well under 10 ms
      on the ESP32-C3 RISC-V at 160 MHz.
  K-Means++ initialization (vs plain K-Means random) was chosen to avoid
  degenerate centroid placement that makes early training epochs unreliable.
}
\end{frame}

\begin{frame}{MCU: ESP32 V1 $\rightarrow$ ESP32-C3 RISC-V}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.46\textwidth}
  \begin{colorblock}[white]{sintefred}{Initial choice}
  \large
  \begin{itemize}
    \item ESP32 DOIT DevKit V1 dual-core
    \item Higher idle current: $\sim$4\,mA (deep sleep)
    \item Oversized for the task
  \end{itemize}
  \end{colorblock}
\end{column}
\begin{column}{0.46\textwidth}
  \begin{colorblock}[white]{maincolor}{Result}
  \large
  \begin{itemize}
    \item ESP32-C3 
    \item $50\,\mu\text{A}$ deep sleep
    \item 20\,mA active
  \end{itemize}
  \end{colorblock}
\end{column}
\end{columns}
\vspace{0.4cm}
\centering
\small Deep-sleep: ESP32-C3 $50\,\mu\text{A}$ vs.\ ESP32 DOIT ${\sim}4\,\text{mA}$; $80{\times}$ reduction in quiescent current
\end{frame}
