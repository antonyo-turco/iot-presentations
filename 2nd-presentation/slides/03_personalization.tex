  \begin{itemize}\setlength{\itemsep}{2pt}\setlength{\parskip}{0pt}
    \item \textbf{ESP32-C3 active:} 20\,\text{mA}
  \item \textbf{Peak current (WiFi / ESP-NOW TX):} 200\,\text{mA}
  \item \textbf{Alarm rate:} 12 alarms/hour = 288 alarms/day
  \item \textbf{Active time per alarm:} 30\,\text{s} $\Rightarrow$ 86\,400\,\text{s/day} total wake time (worst case: 1 alarm/s)
  \item \textbf{Duty cycle:} 10\%
\end{itemize}
