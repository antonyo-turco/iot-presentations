% --- Slide: Prototype and Simulation ---
\begin{frame}{Prototype \& Wokwi Simulation}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
\begin{block}{Development Environment}
Firmware developed with \textbf{ESP-IDF} (not Arduino). Prototype tested on ESP32 DOIT DevKit V1 with Wokwi simulation.
\end{block}
\vspace{0.2cm}
\begin{itemize}
\item \textbf{Wokwi setup}:
  \begin{itemize}
  \item Pushbutton on GPIO35 $\rightarrow$ simulates TLC393 interrupt
  \item Potentiometer on GPIO34 $\rightarrow$ simulates piezo analog signal
  \end{itemize}
\item \textbf{Validated so far}:
  \begin{itemize}
  \item Deep-sleep $\leftrightarrow$ EXT1 wake-up cycle
  \item ADC continuous mode with DMA at 20\,kHz
  \item Average and max amplitude computation
  \item ESP-NOW packet transmission
  \end{itemize}
\end{itemize}
\end{column}

\begin{frame}{Energy Budget Estimation}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
\begin{block}{Battery Life Projection}
Estimating autonomy for a 2000\,mAh Li-Ion cell with 10 alarm events/day.
\end{block}
\vspace{0.2cm}
\renewcommand{\arraystretch}{1.2}
\centering
\begin{tabular}{lr}
\hline
\textbf{State} & \textbf{Current} \\
\hline
Deep sleep (23.98\,h/day) & $50\,\mu$A \\
Active sensing (5\,s $\times$ 10) & ${\sim}\,80$\,mA \\
ESP-NOW TX (${\sim}\,2$\,ms $\times$ 10) & ${\sim}\,130$\,mA \\
\hline
\textbf{Daily avg} & ${\sim}\,66\,\mu$A \\
\hline
\end{tabular}
\renewcommand{\arraystretch}{1.0}

\vspace{0.3cm}
$$T_\text{battery} \approx \frac{2000\,\text{mAh}}{0.066\,\text{mA}} \approx \mathbf{3.5\,\text{years}}$$

{\small With solar trickle charging at ${\sim}\,10$\,mA (even a few hours/day), the system becomes effectively \textbf{perpetual}.}
