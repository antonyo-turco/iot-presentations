\section{Introduction}

\begin{frame}{Problem and Motivation}
\begin{itemize}
\item \textbf{Risk}: residential and small-building entrances are vulnerable to break-in attempts.
\item \textbf{Limits}: wired systems can be invasive, expensive, and difficult to deploy in historical or hard-to-reach locations.
\item \textbf{Vision}: KnockKnock targets a low-cost and scalable alternative with wireless sensors and easy installation.
\item \textbf{Challenge}: reduce false positives while preserving rapid detection.
\end{itemize}
\end{frame}

\begin{frame}{Why IoT and What We Measure}
\begin{itemize}
\item \textbf{Sensing}: nodes collect local vibration/movement signals near doors and windows.
\item \textbf{Data}: events are aggregated and forwarded to a central backend for storage, analysis, and visualization.
\item \textbf{Communication} needs:
      \begin{itemize}
      \item peer-to-peer messaging for fast neighborhood coordination during alarms
      \item publish/subscribe uplink for global monitoring and event history
      \end{itemize}
\item \textbf{Dashboard}: a web interface provides homeowners with a clear and immediate system status view.
\end{itemize}
\end{frame}
