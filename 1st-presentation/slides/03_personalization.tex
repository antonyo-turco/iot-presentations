\section{Personalization}

\footlinecolor{}
\begin{frame}{System Architecture and Response}
\begin{itemize}
\item \textbf{Edge} node behavior:
      \begin{itemize}
      \item idle low-power mode while waiting for passive trigger events
      \item local anomaly scoring based on learned environmental baseline
      \item alarm escalation only when confidence exceeds threshold
      \end{itemize}
\item \textbf{Actions} after alarm confirmation:
      \begin{itemize}
      \item user notification through web dashboard
      \item activation of sirens, floodlights, and optional deadbolt logic
      \item neighborhood node wake-up for richer contextual sensing
      \end{itemize}
\end{itemize}
\end{frame}

\footlinecolor{}
\begin{frame}{Constraints and Design Trade-Offs}
\begin{itemize}
\item \textbf{Bandwidth}:
      \begin{itemize}
      \item support both local alarm propagation and centralized telemetry
      \end{itemize}
\item \textbf{Energy}:
  \begin{itemize}
  \item continuous sensing readiness with strict battery constraints
  \item exploration of indoor solar harvesting feasibility
  \end{itemize}
\item \textbf{Maintenance}:
  \begin{itemize}
  \item no wiring, rapid installation, and modular mesh expansion
  \item operation in hard-to-reach points like high windows or skylights
      \end{itemize}
\item \textbf{Security (secondary)}:
      \begin{itemize}
      \item device identity, secure transport, and resilience against spoofed alerts
      \end{itemize}
\end{itemize}
\end{frame}
