\section{Overview}
\begin{frame}
\frametitle{Project Overview}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.6\textwidth}
\begin{block}{Mission}
Detect intrusion attempts by observing abnormal door and window vibrations together with related physical signals.
\end{block}

\begin{itemize}
\item \textbf{Wireless}: battery-powered nodes with low-impact installation.
\item \textbf{Adaptive}: machine learning learns the normal environmental baseline.
\item \textbf{Actionable}: the system raises alerts and acts in the environment.
\end{itemize}
\end{column}

\begin{column}{0.36\textwidth}
\begin{colorblock}[black]{sintefyellow}{Detection Flow}
\begin{enumerate}
\item Sense
\item Classify
\item Alert
\item Respond
\end{enumerate}
\end{colorblock}
